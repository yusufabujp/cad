৯৬ jquery_14_effect_fade_in_out.html 
<!DOCTYPE html>
<html lang="ja">
<head>
<title>Fade In Out Effect</title>
<meta charset="UTF-8">
<script src="https://ajax.googleapis.com/ajax/libs/jquery/1.12.4/jquery.min.js"></script>
<style>
#div1, #div2, #div3 {
    width:100px;
    height:100px;
    margin: 10px;
    display: none; /* Initially hidden */
}
#div1{ background-color:red; }
#div2{ background-color:green; }
#div3{ background-color:blue; }
</style>
</head>
<body>
<button id="btn1" class="bg-green-500 text-white p-2 rounded mr-2">Fade In</button>
<button id="btn2" class="bg-red-500 text-white p-2 rounded">Fade Out</button>
<div id="div1" class="rounded"></div>
<div id="div2" class="rounded"></div>
<div id="div3" class="rounded"></div>
<script>
$(document).ready(function() {
    $("#btn1").click(function(){
        $("#div1").fadeIn();
        $("#div2").fadeIn("slow");
        $("#div3").fadeIn(3000);
    });
    $("#btn2").click(function(){
        $("#div1").fadeOut();
        $("#div2").fadeOut("slow");
        $("#div3").fadeOut(3000);
    });
});
</script>
</body>
</html>
